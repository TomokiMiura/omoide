24class MenUserCreateView(FormView): 

25 form_class = CustomUserCreationForm 

26 template_name = 'create_men_user.html' 

27 success_url = reverse_lazy('accounts:create_girl_user') 

28 def form_valid(self, form): 

29 # 戻るボタンが押下されたら、入力内容はそのままフォームを再表示 

30 if self.request.POST['next'] == 'back': 

31 return render(self.request, 'create_men_user.html', {'form': form}) 

32 # 確認ボタンが押下されたら、入力内容を表示 

33 elif self.request.POST['next'] == 'confirm': 

34 return render(self.request, 'confirm_men_user.html', {'form': form}) 

35 # 登録ボタンが押下されたら入力内容をフォームに保存 

36 elif self.request.POST['next'] == 'regist': 

37 set_form = form.save() 

38 response = redirect(reverse_lazy('accounts:create_girl_user')) 

39 # Cookieを活用し、一時的に男性ユーザーのインスタンスを保存 

40 response.set_cookie('men_data', set_form) 

41 # CoupleMasterのレコードを作成 

42 CoupleMaster.objects.create(men_id=set_form) 

43 return response 

44 else: 

45 # 通常このルートは通らない 

46 return redirect(reverse_lazy('base:base'))
